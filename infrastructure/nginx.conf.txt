server {
    listen 80;
    
    # Dynamically set from Docker environment variables
    # This allows you to change domains in .env without rebuilding
    server_name ${DOMAIN_NAME} ${API_DOMAIN_NAME} localhost;

    client_max_body_size 50M;

    # Serve React App (The Studio Panel)
    location / {
        root /usr/share/nginx/html;
        index index.html index.htm;
        try_files $uri $uri/ /index.html;
    }

    # Proxy API requests to Node backend
    # This handles both internal Studio requests and external RPC calls
    location /api/ {
        proxy_pass http://api:3000/api/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # Security & Identity Headers
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Note: CORS is now handled primarily by the NodeJS application 
        # to support dynamic Allow-Origin lists from .env
    }
}
